496 void doRender(Window* w) {
497 for (int i = 0; i < w->getProgramListSize(); i++) {
498 unsigned program = w->getProgram(i);
499 glUseProgram(program);
500
501 switch (projectionMode_) {
502 case ProjectionMode::kPerspective:
503 glm::mat4 perspective = getPerspectiveMatrix(60.0f, 1024.0f / 768.0f, 0.01f, 100000.0f);
504
505 glUniformMatrix4fv(glGetUniformLocation(program, "u_p_matrix"), 1, GL_FALSE, glm::value_ptr(perspective));
506 break;
507 case ProjectionMode::kOrthogonal:
508 glm::mat4 ortographic = getOrthogonalMatrix(-1.0f, 1.0f, -1.0f, 1.0f, 0.01f, 100000.0f);
509
510 glUniformMatrix4fv(glGetUniformLocation(program, "u_p_matrix"), 1, GL_FALSE, glm::value_ptr(ortographic));
511 break;
512 }
513 //View
514 glm::mat4 view = getViewMatrix(pos_ + forward_, up_);
515 GLint viewMatrixLoc = glGetUniformLocation(program, "u_v_matrix");
516 glUniformMatrix4fv(viewMatrixLoc, 1, GL_FALSE, glm::value_ptr(view));
517
518 //Camera position
519 GLint camPosLoc = glGetUniformLocation(program, "u_camera_pos");
520 glUniform3f(camPosLoc, pos_.x, pos_.y, pos_.z);
521 }
